. Content-to-Code Handoff
5.1 Component List & Props
Navbar: props: menuLinks (array of { name, href }), logo.
Hero: props: title, subtitle, backgroundImage, ctaButtons (array of { text, href }).
About: props: profileImage, introText (string/JSX), statItems (array of { label, value }).
Timeline: props: items (array of { date, title, company, description, logoSrc }).
BusinessImpact: props: impactItems (array of { icon, title, description }).
ServiceCard: props: icon, serviceName, description.
ProjectsGrid: contains multiple ProjectCard.
ProjectCard: props: name, tagline, shortDesc, features (array), techStack (array), caseStudyUrl.
CaseStudySection: props: projectName, problem, solution, features, techStack, value, ctaText, ctaLink.
Testimonials: props: quotes (array of { text, author }), logos (array of logo image URLs).
ContactForm: controlled component with fields name, email, message.
Footer: props: links (array), socialLinks (array of icons/URLs).

// components/Hero.jsx
export default function Hero({ title, subtitle, ctaButtons }) {
  return (
    <section className="relative flex flex-col md:flex-row items-center h-screen bg-darknavy text-white px-8 md:px-16">
      <div className="flex-1 space-y-6">
        <h1 className="text-5xl md:text-6xl font-bold">{title}</h1>
        <p className="text-lg md:text-xl">{subtitle}</p>
        <div className="space-x-4">
          {ctaButtons.map(btn => (
            <a
              key={btn.text}
              href={btn.href}
              className="inline-block bg-blue-500 hover:bg-blue-600 text-white font-semibold py-3 px-6 rounded-lg transition"
            >
              {btn.text}
            </a>
          ))}
        </div>
      </div>
      <div className="hidden md:block flex-1">
        <img src="/images/ashik-profile.jpg" alt="Ashikur Rahaman" className="rounded-full shadow-lg" />
      </div>
    </section>
  )
}

ProjectCard Component
jsx

// components/ProjectCard.jsx
export default function ProjectCard({ name, tagline, features, techStack, imageUrl, caseStudyUrl }) {
  return (
    <div className="bg-gradient-to-br from-gray-800 to-gray-900 rounded-xl p-6 shadow-md flex flex-col">
      <img src={imageUrl} alt={`${name} screenshot`} className="w-full rounded-md mb-4" />
      <h3 className="text-xl font-semibold text-white">{name}</h3>
      <p className="text-gray-300 mb-4">{tagline}</p>
      <ul className="list-disc list-inside text-gray-400 flex-1">
        {features.map((feat, i) => <li key={i}>{feat}</li>)}
      </ul>
      <div className="mt-4">
        <p className="text-sm text-gray-500">Tech: {techStack.join(', ')}</p>
        <a href={caseStudyUrl} className="mt-3 inline-block bg-blue-600 hover:bg-blue-700 text-white py-2 px-4 rounded-lg">
          View Case Study
        </a>
      </div>
    </div>
  )
}
